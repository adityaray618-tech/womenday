    <script>
        const video = document.getElementById('webcam-video');
        const unlockBtn = document.getElementById('unlock-btn');
        const initScreen = document.getElementById('mirror-init');
        const textOverlay = document.getElementById('text-overlay');
        const dynamicWord = document.getElementById('dynamic-word');
        const captureBtn = document.getElementById('capture-btn');
        const celebrationArea = document.getElementById('celebration-area');
        const flashOverlay = document.getElementById('flash-overlay');
        const canvas = document.getElementById('capture-canvas');
        const letterModal = document.getElementById('letter-modal');

        const phrases = [
            "The Bravest Woman",
            "The Strongest Heart",
            "A True Masterpiece",
            "Unstoppable Spirit",
            "Pure Brilliance",
            "Simply Beautiful",
            "A Force of Nature"
        ];

        const templates = [
            { name: "Gold Classic", border: ["#cf9f2d", "#fff0b3", "#8a6316"], text: "#fbbf24" },
            { name: "Rose Quartz", border: ["#ff9a9e", "#fecfef", "#ff758c"], text: "#fbc2eb" },
            { name: "Royal Purple", border: ["#663399", "#a855f7", "#3b0764"], text: "#d8b4fe" },
            { name: "Diamond Shine", border: ["#e2e8f0", "#ffffff", "#94a3b8"], text: "#ffffff" }
        ];

        unlockBtn.addEventListener('click', async () => {
            try {
                unlockBtn.innerText = "Opening Mirror...";
                const stream = await navigator.mediaDevices.getUserMedia({ video: true });
                video.srcObject = stream;
                
                video.onplay = () => {
                    initScreen.classList.add('hidden');
                    video.style.opacity = "1";
                    captureBtn.classList.remove('hidden');
                    celebrationArea.classList.remove('hidden');
                    startEmpowermentLoop();
                };
            } catch (err) {
                alert("Please allow camera access to see your reflection!");
            }
        });

        function startEmpowermentLoop() {
            let i = 0;
            textOverlay.classList.add('show');
            dynamicWord.innerText = phrases[i];

            setInterval(() => {
                textOverlay.classList.remove('show');
                setTimeout(() => {
                    i = (i + 1) % phrases.length;
                    dynamicWord.innerText = phrases[i];
                    textOverlay.classList.add('show');
                }, 800);
            }, 4000);
        }

        captureBtn.addEventListener('click', () => {
            const template = templates[Math.floor(Math.random() * templates.length)];
            
            flashOverlay.style.animation = 'none';
            void flashOverlay.offsetWidth;
            flashOverlay.style.animation = 'flash-anim 0.5s ease-out';

            const ctx = canvas.getContext('2d');
            
            // Mirror Frame dimensions
            const frameThickness = 40;
            canvas.width = video.videoWidth + (frameThickness * 2);
            canvas.height = video.videoHeight + (frameThickness * 2);

            // 1. Draw the Background Mirror Frame
            const grad = ctx.createLinearGradient(0, 0, canvas.width, canvas.height);
            grad.addColorStop(0, template.border[0]);
            grad.addColorStop(0.5, template.border[1]);
            grad.addColorStop(1, template.border[2]);
            ctx.fillStyle = grad;
            ctx.fillRect(0, 0, canvas.width, canvas.height);

            // 2. Draw the Video Reflection (Mirrored)
            ctx.save();
            ctx.translate(canvas.width - frameThickness, frameThickness);
            ctx.scale(-1, 1);
            ctx.drawImage(video, 0, 0, video.videoWidth, video.videoHeight);
            ctx.restore();

            // 3. Add Message Bar
            ctx.fillStyle = "rgba(0, 0, 0, 0.6)";
            ctx.fillRect(frameThickness, canvas.height - frameThickness - 120, video.videoWidth, 120);
            
            // 4. Draw the Dynamic Text
            ctx.font = "bold 38px Cinzel, serif";
            ctx.fillStyle = template.text;
            ctx.textAlign = "center";
            ctx.shadowColor = "rgba(0,0,0,0.5)";
            ctx.shadowBlur = 10;
            ctx.fillText(dynamicWord.innerText, canvas.width / 2, canvas.height - frameThickness - 60);
            ctx.shadowBlur = 0;

            // 5. Add Website Branding Watermark
            ctx.font = "italic 18px Montserrat";
            ctx.fillStyle = "rgba(255,255,255,0.9)";
            ctx.fillText("INTERNATIONAL WOMEN'S DAY 2024", canvas.width / 2, frameThickness / 1.5);

            // Download the final "Mirror Picture"
            const image = canvas.toDataURL("image/png");
            const link = document.createElement('a');
            link.download = `WomensDay-Gift-${Date.now()}.png`;
            link.href = image;
            document.body.appendChild(link);
            link.click();
            document.body.removeChild(link);
        });

        document.getElementById('open-letter-btn').onclick = () => {
            letterModal.classList.remove('pointer-events-none');
            letterModal.classList.add('opacity-100');
        };

        document.getElementById('close-letter').onclick = () => {
            letterModal.classList.add('pointer-events-none');
            letterModal.classList.remove('opacity-100');
        };
    </script>
